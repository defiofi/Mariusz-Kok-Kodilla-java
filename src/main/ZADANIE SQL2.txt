SELECT ISSUES.SUMMARY , ISSUESLISTS.NAME FROM ISSUES
JOIN ISSUESLISTS ON ISSUES.ISSUESLIST_ID = ISSUESLISTS.ID;

SELECT ISSUES.SUMMARY , USERS.FIRSTNAME , USERS.LASTNAME FROM ISSUES
JOIN USERS ON USERS.ID = ISSUES.USER_ID_ASSIGNEDTO
ORDER BY ISSUES.ID;

SELECT USERS.FIRSTNAME , USERS.LASTNAME , COUNT(*) AS ASSIGNED_TASKS FROM USERS
JOIN ISSUES ON USERS.ID = ISSUES.USER_ID_ASSIGNEDTO
GROUP BY USERS.FIRSTNAME;
